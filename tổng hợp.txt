-- [[ CẤU HÌNH BẬT TẮT ]] --
_G.PVP_Settings = {
    EspLine = true -- Đổi thành false để tắt
}

-- [[ HÀM VẼ ESP LINE TỪ ĐỈNH MÀN HÌNH ]] --
local function CreateESPLine(target)
    local Line = Drawing.new("Line")
    
    RunService.RenderStepped:Connect(function()
        if target.Character and target.Character:FindFirstChild("HumanoidRootPart") and target.Character.Humanoid.Health > 0 then
            local RootPart = target.Character.HumanoidRootPart
            local Pos, OnScreen = Camera:WorldToViewportPoint(RootPart.Position)
            
            if OnScreen and _G.PVP_Settings.EspLine then
                Line.Visible = true
                -- Điểm bắt đầu: Giữa cạnh trên của màn hình
                Line.From = Vector2.new(Camera.ViewportSize.X / 2, 0) 
                -- Điểm kết thúc: Vị trí đối thủ
                Line.To = Vector2.new(Pos.X, Pos.Y)
                Line.Color = Color3.fromRGB(0, 255, 255) -- Màu xanh Neon
                Line.Thickness = 1
                Line.Transparency = 1
            else
                Line.Visible = false
            end
        else
            Line.Visible = false
        end
    end)
end

-- Tự động áp dụng cho tất cả người chơi trong server
for _, v in pairs(Players:GetPlayers()) do
    if v ~= player then
        CreateESPLine(v)
    end
end

-- Áp dụng cho người chơi mới tham gia
Players.PlayerAdded:Connect(function(v)
    if v ~= player then
        CreateESPLine(v)
    end
end)
}

                pcall(function()

end)

-- [[ 2. HÀM TẠO NÚT BẬT/TẮT TRÊN GIAO DIỆN ]] --
local function CreateToggle(name, parent, callback)
    local toggleFrame = Instance.new("TextButton")
    toggleFrame.Size = UDim2.new(1, -10, 0, 40)
    toggleFrame.BackgroundColor3 = Color3.fromRGB(30, 33, 45)
    toggleFrame.Text = "  " .. name
    toggleFrame.TextColor3 = Color3.new(1, 1, 1)
    toggleFrame.TextXAlignment = Enum.TextXAlignment.Left
    toggleFrame.Font = Enum.Font.Gotham
    toggleFrame.Parent = parent
    
    local corner = Instance.new("UICorner", toggleFrame)
    corner.CornerRadius = UDim.new(0, 8)

    -- Trạng thái nội bộ
    local active = false

    toggleFrame.MouseButton1Click:Connect(function()
        active = not active
        
        -- Hiệu ứng đổi màu khi bật/tắt
        if active then
            toggleFrame.BackgroundColor3 = Color3.fromRGB(0, 180, 255) -- Màu xanh khi bật
            toggleFrame.Text = "  [ ON ] " .. name
        else
            toggleFrame.BackgroundColor3 = Color3.fromRGB(30, 33, 45) -- Màu tối khi tắt
            toggleFrame.Text = "  [ OFF ] " .. name
        end
        
        -- Chạy chức năng
        callback(active)
    end)
end
