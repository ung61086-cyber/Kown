-- [[ HÀM TÌM QUÁI GẦN NHẤT ]] --
local function GetMonster()
    local target = nil
    local dist = math.huge
    for _, v in pairs(workspace.Enemies:GetChildren()) do
        if v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 and v:FindFirstChild("HumanoidRootPart") then
            local d = (player.Character.HumanoidRootPart.Position - v.HumanoidRootPart.Position).Magnitude
            if d < dist then
                dist = d
                target = v
            end
        end
    end
    return target
end

-- [[ HÀM DỊCH CHUYỂN AN TOÀN ]] --
local function TP(cf)
    if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        local distance = (player.Character.HumanoidRootPart.Position - cf.Position).Magnitude
        local speed = 350 -- Tốc độ bay chuẩn
        TweenService:Create(player.Character.HumanoidRootPart, TweenInfo.new(distance/speed, Enum.EasingStyle.Linear), {CFrame = cf}):Play()
    end
end

-- [[ NÚT BẤM (V) ]] --
local function addToggle(name, callback)
    local f = Instance.new("Frame", scroll)
    f.Size = UDim2.new(1, -5, 0, 45)
    f.BackgroundColor3 = Color3.fromRGB(25, 30, 45)
    f.BackgroundTransparency = 0.5
    Instance.new("UICorner", f).CornerRadius = UDim.new(0, 20)

    local btn = Instance.new("TextButton", f)
    btn.Size = UDim2.new(0, 28, 0, 28); btn.Position = UDim2.new(1, -38, 0.5, -14)
    btn.Text = ""; btn.BackgroundColor3 = Color3.fromRGB(40, 45, 60)
    Instance.new("UICorner", btn).CornerRadius = UDim.new(1, 0)

    local l = Instance.new("TextLabel", f)
    l.Text = name; l.Size = UDim2.new(1,-60,1,0); l.Position = UDim2.new(0,15,0,0)
    l.TextColor3 = Color3.new(1,1,1); l.BackgroundTransparency = 1; l.TextXAlignment = 0

    local active = false
    btn.MouseButton1Click:Connect(function()
        active = not active
        btn.Text = active and "✓" or ""
        btn.BackgroundColor3 = active and Color3.fromRGB(0, 200, 255) or Color3.fromRGB(40, 45, 60)
        callback(active)
    end)
end

-- [[ LẤP ĐẦY CHỨC NĂNG ]] --

-- 1. Auto Farm Level
addToggle("Auto Farm Level", function(v)
    _G.Config.AutoFarm = v
    task.spawn(function()
        while _G.Config.AutoFarm do
            pcall(function()
                local monster = GetMonster()
                if monster then
                    -- Bay tới quái
                    player.Character.HumanoidRootPart.CFrame = monster.HumanoidRootPart.CFrame * CFrame.new(0, _G.Config.Distance, 0)
                    -- Auto Click
                    game:GetService("VirtualUser"):CaptureController()
                    game:GetService("VirtualUser"):Button1Down(Vector2.new(1280, 672))
                end
            end)
            task.wait()
        end
    end)
end)

-- 2. Fast Attack (No Animation)
addToggle("Fast Attack (S1)", function(v)
    _G.Config.FastAttack = v
    task.spawn(function()
        while _G.Config.FastAttack do
            pcall(function()
                local combat = player.Character:FindFirstChildOfClass("Tool")
                if combat then combat:Activate() end
            end)
            task.wait(0.05)
        end
    end)
end)
