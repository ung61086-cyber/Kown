-- [[ HÀM TẠO NÚT TÍCH (V) - GIỐNG ẢNH 1 ]] --
local function addToggle(name, callback)
    local f = Instance.new("Frame", scroll)
    f.Size = UDim2.new(1, -10, 0, 45)
    f.BackgroundColor3 = Color3.fromRGB(25, 30, 45)
    f.BackgroundTransparency = 0.5
    Instance.new("UICorner", f).CornerRadius = UDim.new(0, 22)

    local l = Instance.new("TextLabel", f)
    l.Size = UDim2.new(1, -60, 1, 0); l.Position = UDim2.new(0, 20, 0, 0)
    l.Text = name; l.TextColor3 = Color3.new(1,1,1); l.BackgroundTransparency = 1; l.TextXAlignment = 0
    l.Font = Enum.Font.Gotham

    local btn = Instance.new("TextButton", f)
    btn.Size = UDim2.new(0, 30, 0, 30); btn.Position = UDim2.new(1, -40, 0.5, -15)
    btn.BackgroundColor3 = Color3.fromRGB(40, 45, 60); btn.Text = ""
    Instance.new("UICorner", btn).CornerRadius = UDim.new(1, 0)

    local active = false
    btn.MouseButton1Click:Connect(function()
        active = not active
        btn.BackgroundColor3 = active and Color3.fromRGB(60, 120, 255) or Color3.fromRGB(40, 45, 60)
        btn.Text = active and "✓" or ""
        callback(active)
    end)
end

-- [[ HÀM DỊCH CHUYỂN (TWEEN) ]] --
local function Teleport(cf)
    local char = player.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        local p1 = char.HumanoidRootPart.Position
        local p2 = cf.Position
        local dist = (p1 - p2).Magnitude
        TweenService:Create(char.HumanoidRootPart, TweenInfo.new(dist/250, Enum.EasingStyle.Linear), {CFrame = cf}):Play()
    end
end

-- [[ HÀM ESP (S1, S2, S3) ]] --
local function ApplyESP(v)
    if v == player then return end
    local Line = Drawing.new("Line")
    local Box = Drawing.new("Square")
    local Name = Drawing.new("Text")

    RunService.RenderStepped:Connect(function()
        if _G.ESP_Enabled and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
            local Pos, OnScreen = camera:WorldToViewportPoint(v.Character.HumanoidRootPart.Position)
            if OnScreen then
                -- Line S1
                Line.Visible = true
                Line.From = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y)
                Line.To = Vector2.new(Pos.X, Pos.Y)
                Line.Color = Color3.fromRGB(0, 255, 150)
                Line.Thickness = 1
                -- Box S2
                Box.Visible = true
                Box.Size = Vector2.new(2000/Pos.Z, 3000/Pos.Z)
                Box.Position = Vector2.new(Pos.X - Box.Size.X/2, Pos.Y - Box.Size.Y/2)
                Box.Color = Color3.fromRGB(0, 255, 150)
                -- Name S3
                Name.Visible = true
                Name.Text = v.Name .. " [Lv." .. (v:FindFirstChild("Data") and v.Data.Level.Value or "???") .. "]"
                Name.Position = Vector2.new(Pos.X, Pos.Y - Box.Size.Y/2 - 15)
                Name.Center = true; Name.Outline = true; Name.Color = Color3.new(1,1,1)
            else
                Line.Visible = false; Box.Visible = false; Name.Visible = false
            end
        else
            Line.Visible = false; Box.Visible = false; Name.Visible = false
        end
    end)
end
