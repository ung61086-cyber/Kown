-- [[ 4. HÀM TẠO TOGGLE ]] --
local function CreateToggle(name, callback)
    local btn = Instance.new("TextButton", scroll)
    btn.Size = UDim2.new(1, -5, 0, 40)
    btn.BackgroundColor3 = Color3.fromRGB(30, 33, 45)
    btn.Text = "[ OFF ] " .. name
    btn.TextColor3 = Color3.new(1, 1, 1)
    btn.Font = Enum.Font.Gotham
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 8)

    local active = false
    btn.MouseButton1Click:Connect(function()
        active = not active
        btn.BackgroundColor3 = active and Color3.fromRGB(0, 180, 255) or Color3.fromRGB(30, 33, 45)
        btn.Text = active and "[ ON ] " .. name or "[ OFF ] " .. name
        callback(active)
    end)
end

-- [[ 5. LOGIC CHỨC NĂNG TỔNG HỢP (QUAN TRỌNG) ]] --

-- Cập nhật Server Stats
local function UpdateStats()
    serverStats.Text = "SERVER: " .. #Players:GetPlayers() .. "/" .. Players.MaxPlayers .. " Players"
end
Players.PlayerAdded:Connect(UpdateStats)
Players.PlayerRemoving:Connect(UpdateStats)
UpdateStats()

-- ESP Line
local function CreateESPLine(target)
    local Line = Drawing.new("Line")
    RunService.RenderStepped:Connect(function()
        if target.Character and target.Character:FindFirstChild("HumanoidRootPart") and target.Character.Humanoid.Health > 0 then
            local Pos, OnScreen = Camera:WorldToViewportPoint(target.Character.HumanoidRootPart.Position)
            if OnScreen and _G.PVP_Settings.EspLine then
                Line.Visible = true
                Line.From = Vector2.new(Camera.ViewportSize.X / 2, 0)
                Line.To = Vector2.new(Pos.X, Pos.Y)
                Line.Color = Color3.new(0, 1, 1)
                Line.Thickness = 1
            else Line.Visible = false end
        else Line.Visible = false end
    end)
end
for _, v in pairs(Players:GetPlayers()) do if v ~= player then CreateESPLine(v) end end
Players.PlayerAdded:Connect(function(v) if v ~= player then CreateESPLine(v) end end)

-- VÒNG LẶP CHÍNH (Hitbox, TP, FOV, Circle)
RunService.RenderStepped:Connect(function()
    -- Giữ FOV
    Camera.FieldOfView = _G.PVP_Settings.FOV_Value

    -- Logic Vòng Tròn Aim
    if _G.PVP_Settings.AimCircle and _G.PVP_Settings.CircleRadius > 0 then
        AimCircle.Visible = true
        AimCircle.Radius = _G.PVP_Settings.CircleRadius
        AimCircle.Position = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
    else
        AimCircle.Visible = false
    end

    -- Hitbox & Tìm mục tiêu TP
    local target, dist = nil, math.huge
    for _, v in pairs(Players:GetPlayers()) do
        if v ~= player and v.Character and v.Character:FindFirstChild("HumanoidRootPart") and v.Character.Humanoid.Health > 0 then
            -- Hitbox
            if _G.PVP_Settings.HitboxEnabled then
                local hrp = v.Character.HumanoidRootPart
                hrp.Size = Vector3.new(_G.PVP_Settings.HitboxSize, _G.PVP_Settings.HitboxSize, _G.PVP_Settings.HitboxSize)
                hrp.Transparency = 0.7
                hrp.CanCollide = false
            end
            -- Tìm khoảng cách
            local d = (player.Character.HumanoidRootPart.Position - v.Character.HumanoidRootPart.Position).Magnitude
            if d < dist then dist = d; target = v end
        end
    end

    -- TP Kill
    if _G.PVP_Settings.TP_Kill and target then
        player.Character.HumanoidRootPart.CFrame = target.Character.HumanoidRootPart.CFrame * CFrame.new(_G.PVP_Settings.TP_Offset)
    end
end)

        local hum = player.Character.Humanoid

        healthFill.BackgroundColor3 = hp > 0.6 and Color3.new(0,1,0) or hp > 0.3 and Color3.new(1,1,0) or Color3.new(1,0,0)
    end
end)

CreateToggle("Chỉnh Vòng Aim (0-100)", function(state)
    _G.PVP_Settings.AimCircle = state
    if state then
        if _G.PVP_Settings.CircleRadius >= 100 then
            _G.PVP_Settings.CircleRadius = 0
        else
            _G.PVP_Settings.CircleRadius = _G.PVP_Settings.CircleRadius + 25 -- Mỗi lần bấm tăng 25 đơn vị
        end
    else
        _G.PVP_Settings.CircleRadius = 0
    end
end)
