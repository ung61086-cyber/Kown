-- [[ 3. HỆ THỐNG VẼ ESP BOX VUÔNG & VẠCH KẺ XANH ]] --
local function CreateESP(plr)
    local Line = Drawing.new("Line")
    local Box = Drawing.new("Square")
    local Health = Drawing.new("Text")

    RunService.RenderStepped:Connect(function()
        if plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") and plr ~= LocalPlayer then
            local HRP = plr.Character.HumanoidRootPart
            local Pos, OnScreen = Camera:WorldToViewportPoint(HRP.Position)

            if OnScreen then
                -- VẠCH KẺ XANH
                Line.Visible = _G.Config.ESP_Line
                Line.From = Vector2.new(Camera.ViewportSize.X / 2, 0)
                Line.To = Vector2.new(Pos.X, Pos.Y - (3500 / Pos.Z) / 2)
                Line.Color = Global_Green
                Line.Thickness = 1.5

                -- KHUNG VUÔNG BAO QUANH XANH
                if _G.Config.ESP_Box then
                    local SizeX = 2500 / Pos.Z
                    local SizeY = 3500 / Pos.Z
                    Box.Visible = true
                    Box.Size = Vector2.new(SizeX, SizeY)
                    Box.Position = Vector2.new(Pos.X - SizeX / 2, Pos.Y - SizeY / 2)
                    Box.Color = Global_Green
                    Box.Thickness = 2
                    Box.Filled = false
                else Box.Visible = false end

                -- SOI MÁU XANH
                Health.Visible = _G.Config.ESP_HP
                local hum = plr.Character:FindFirstChild("Humanoid")
                Health.Text = hum and "HP: " .. math.floor(hum.Health) or "HP: 0"
                Health.Position = Vector2.new(Pos.X, Pos.Y + (3500 / Pos.Z) / 2 + 5)
                Health.Color = Global_Green
                Health.Center = true; Health.Outline = true
            else Line.Visible = false; Box.Visible = false; Health.Visible = false end
        else Line.Visible = false; Box.Visible = false; Health.Visible = false end
    end)
end

for _, v in pairs(Players:GetPlayers()) do CreateESP(v) end
Players.PlayerAdded:Connect(CreateESP)

-- [[ 4. VÒNG TRÒN FOV XANH & AIMBOT CỰC MẠNH ]] --
local FOVCircle = Drawing.new("Circle")
FOVCircle.Thickness = 1.5; FOVCircle.Color = Global_Green; FOVCircle.Filled = false

local function GetClosest()
    local target = nil; local dist = _G.Config.FOVSize
    for _, v in pairs(Players:GetPlayers()) do
        if v ~= LocalPlayer and v.Character and v.Character:FindFirstChild("Head") then
            local pos, vis = Camera:WorldToViewportPoint(v.Character.Head.Position)
            if vis then
                local mag = (Vector2.new(pos.X, pos.Y) - Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)).Magnitude
                if mag < dist then target = v; dist = mag end
            end
        end
    end
    return target
end

RunService.RenderStepped:Connect(function()
    FOVCircle.Visible = _G.Config.ShowFOV; FOVCircle.Radius = _G.Config.FOVSize; FOVCircle.Position = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
    if _G.Config.Aimbot then
        local t = GetClosest()
        if t then Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, t.Character.Head.Position), _G.Config.Smoothness) end
    end
    if _G.Config.Hitbox then
        for _, v in pairs(Players:GetPlayers()) do
            if v ~= LocalPlayer and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
                v.Character.HumanoidRootPart.Size = Vector3.new(_G.Config.HitboxSize, _G.Config.HitboxSize, _G.Config.HitboxSize)
                v.Character.HumanoidRootPart.Transparency = 0.7
            end
        end
    end
end)
