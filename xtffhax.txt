-- XTFFHAX STYLE DEMO MENU (UI ONLY)

local Players = game:GetService("Players")
local plr = Players.LocalPlayer

local gui = Instance.new("ScreenGui", plr.PlayerGui)
gui.Name = "XTFFHAX_DEMO"

local main = Instance.new("Frame", gui)
main.Size = UDim2.fromOffset(420, 240)
main.Position = UDim2.fromScale(0.5, 0.5)
main.AnchorPoint = Vector2.new(0.5, 0.5)
main.BackgroundColor3 = Color3.fromRGB(20, 24, 32)
main.BorderSizePixel = 0
main.Visible = true
Instance.new("UICorner", main).CornerRadius = UDim.new(0, 10)

-- Tabs
local tabs = {"PLAYER","AIM","SETTINGS"}
local tabBtns = {}
local pages = {}

for i,name in ipairs(tabs) do
    local b = Instance.new("TextButton", main)
    b.Size = UDim2.fromOffset(110, 36)
    b.Position = UDim2.fromOffset(10, 10 + (i-1)*46)
    b.Text = name
    b.BackgroundColor3 = Color3.fromRGB(40,40,40)
    b.TextColor3 = Color3.new(1,1,1)
    b.BorderSizePixel = 0
    Instance.new("UICorner", b).CornerRadius = UDim.new(0, 8)
    tabBtns[name] = b

    local page = Instance.new("Frame", main)
    page.Size = UDim2.fromOffset(270, 200)
    page.Position = UDim2.fromOffset(140, 20)
    page.BackgroundTransparency = 1
    page.Visible = false
    pages[name] = page

    b.MouseButton1Click:Connect(function()
        for _,p in pairs(pages) do p.Visible = false end
        for _,t in pairs(tabBtns) do t.BackgroundColor3 = Color3.fromRGB(40,40,40) end
        page.Visible = true
        b.BackgroundColor3 = Color3.fromRGB(60,120,255)
    end)
end

-- Toggle
local function createToggle(parent, text, y)
    local btn = Instance.new("TextButton", parent)
    btn.Size = UDim2.fromOffset(240, 36)
    btn.Position = UDim2.fromOffset(0, y)
    btn.Text = text .. ": OFF"
    btn.BackgroundColor3 = Color3.fromRGB(60,0,0)
    btn.TextColor3 = Color3.new(1,1,1)
    btn.BorderSizePixel = 0
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 8)

    local state = false
    btn.MouseButton1Click:Connect(function()
        state = not state
        if state then
            btn.Text = text .. ": ON"
            btn.BackgroundColor3 = Color3.fromRGB(0,120,0)
        else
            btn.Text = text .. ": OFF"
            btn.BackgroundColor3 = Color3.fromRGB(60,0,0)
        end
    end)
end

-- Pages
createToggle(pages.PLAYER, "Draw Health", 0)
createToggle(pages.PLAYER, "Draw Name", 45)
createToggle(pages.PLAYER, "Draw Distance", 90)
createToggle(pages.PLAYER, "Alert 360", 135)

createToggle(pages.AIM, "Aimbot", 0)
createToggle(pages.AIM, "Aimbot Knock", 45)
createToggle(pages.AIM, "HideBot", 90)

createToggle(pages.SETTINGS, "Show FPS", 0)
createToggle(pages.SETTINGS, "Reduce Lag", 45)

-- Open default tab
pages.PLAYER.Visible = true
tabBtns.PLAYER.BackgroundColor3 = Color3.fromRGB(60,120,255)

-- Nút ẩn/hiện
local toggleBtn = Instance.new("TextButton", gui)
toggleBtn.Size = UDim2.fromOffset(50, 50)
toggleBtn.Position = UDim2.fromOffset(20, 200)
toggleBtn.Text = "XT"
toggleBtn.Font = Enum.Font.GothamBold
toggleBtn.TextSize = 16
toggleBtn.TextColor3 = Color3.new(1,1,1)
toggleBtn.BackgroundColor3 = Color3.fromRGB(0, 140, 255)
toggleBtn.BorderSizePixel = 0
Instance.new("UICorner", toggleBtn).CornerRadius = UDim.new(1, 0)

local open = true
toggleBtn.MouseButton1Click:Connect(function()
    open = not open
    main.Visible = open
end)
