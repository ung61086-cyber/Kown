-- MENU iOS DEMO - CHẮC CHẮN LÊN
local UIS = game:GetService("UserInputService")
local Player = game.Players.LocalPlayer
local Lighting = game:GetService("Lighting")

-- GUI
local gui = Instance.new("ScreenGui")
gui.Name = "IOSMenuDemo"
gui.ResetOnSpawn = false
gui.Parent = Player:WaitForChild("PlayerGui")

local main = Instance.new("Frame", gui)
main.Size = UDim2.fromScale(0.45, 0.4)
main.Position = UDim2.fromScale(0.275, 0.3)
main.BackgroundColor3 = Color3.fromRGB(25,25,28)
main.Visible = false
main.Active = true
main.Draggable = true
Instance.new("UICorner", main).CornerRadius = UDim.new(0,20)

-- Title
local title = Instance.new("TextLabel", main)
title.Size = UDim2.fromScale(1,0.2)
title.Text = "MIGUL iOS"
title.Font = Enum.Font.GothamBold
title.TextScaled = true
title.TextColor3 = Color3.fromRGB(0,170,255)
title.BackgroundTransparency = 1

-- Button
local btn = Instance.new("TextButton", main)
btn.Size = UDim2.fromScale(0.8,0.25)
btn.Position = UDim2.fromScale(0.1,0.45)
btn.Text = "Toggle FullBright"
btn.Font = Enum.Font.Gotham
btn.TextScaled = true
btn.TextColor3 = Color3.new(1,1,1)
btn.BackgroundColor3 = Color3.fromRGB(45,45,50)
Instance.new("UICorner", btn).CornerRadius = UDim.new(0,14)

-- FullBright
local on = false
btn.MouseButton1Click:Connect(function()
	on = not on
	if on then
		Lighting.Brightness = 3
		Lighting.ClockTime = 14
		btn.Text = "FullBright: ON"
	else
		Lighting.Brightness = 1
		btn.Text = "FullBright: OFF"
	end
end)

-- Toggle menu
UIS.InputBegan:Connect(function(i,gp)
	if gp then return end
	if i.KeyCode == Enum.KeyCode.RightShift then
		main.Visible = not main.Visible
	end
end)
